Ejemplo 1
var foo = 1;
function bar() {
	if (!foo) {
		var foo = 10;
	}
	alert(foo);
}
bar();
	
Ejemplo 2
var a = 1;
function b() {
	a = 10;
	return;
	function a() {}
}
b();
alert(a);

function test() {
	foo(); // TypeError "foo is not a function"
	bar(); // "this will run!"
	var foo = function () { // function expression assigned to local variable 'foo'
		alert("this won't run!");
	}
	function bar() { // function declaration, given the name 'bar'
		alert("this will run!");
	}
}
test();

function test() {
	bar(); // "this still will run!"
	var foo = function () { // function expression assigned to local variable 'foo'
		alert("this won't run!");

}
